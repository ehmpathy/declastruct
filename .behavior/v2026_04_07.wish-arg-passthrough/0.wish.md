wish =

                                                                                       
  declastruct enhancement: arg passthrough
                                                                                                                                           
  wish: pass custom args to wish files via -- separator             
                                                                                                                                           
  usage:
  npx declastruct plan --wish resources.ts --into plan.json -- --env test --verbose                                                        
                                                                    
  behavior:                                                                                                                                
  - args after -- are passed to getProviders() and getResources() as second parameter
  - or appended to process.argv before importing the wish file                                                                             
                                                                    
  example resources file:                                                                                                                  
  export const getResources = async (args?: string[]) => {          
    const env = args?.includes('--env')                                                                                                    
      ? args[args.indexOf('--env') + 1]                                                                                                    
      : 'test';                        
    // ...                                                                                                                                 
  };                                                                
                                                                                                                                           
  alternative: declastruct could parse known flags like --env and set them as environment variables before loading the wish file. simpler
  for common cases.
