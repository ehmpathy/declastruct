# blackbox criteria: arg passthrough

> **root motto:** don't let declastruct interfere with --arg passthrough

---

# usecase.1 = plan with passthrough args

given('a wish file that parses --env from process.argv')
  when('user runs: npx declastruct plan --wish resources.ts --into plan.json -- --env prod')
    then('wish file sees --env prod in process.argv')
      sothat('wish file can configure resources for prod environment')
    then('plan is generated with prod configuration')
    then('-- separator is stripped from process.argv')
      sothat('wish file receives clean args')

  when('user runs: npx declastruct plan --wish resources.ts --into plan.json -- --env prod --region us-west-2')
    then('wish file sees both --env prod and --region us-west-2 in process.argv')
      sothat('multiple passthrough args work together')

---

# usecase.2 = plan without passthrough args

given('a wish file that parses args with defaults')
  when('user runs: npx declastruct plan --wish resources.ts --into plan.json')
    then('wish file sees empty args in process.argv (only node and entry paths)')
      sothat('wish file falls back to defaults')
    then('plan is generated with default configuration')

---

# usecase.3 = apply ignores passthrough args

given('a plan file generated with --env test')
  when('user runs: npx declastruct apply --plan plan.json')
    then('apply uses the state captured in plan file')
      sothat('plan/apply contract is honored')

  when('user runs: npx declastruct apply --plan plan.json -- --env prod')
    then('--env prod is ignored')
    then('apply uses the state captured in plan file')
      sothat('apply cannot diverge from planned state')

---

# usecase.4 = help text discoverability

given('user wants to learn about passthrough args')
  when('user runs: npx declastruct plan --help')
    then('help output shows [-- <wish-args>] in usage')
      sothat('passthrough args are discoverable')

---

# usecase.5 = unknown option guidance

given('user passes unknown flag without -- separator')
  when('user runs: npx declastruct plan --wish resources.ts --into plan.json --env prod')
    then('declastruct errors with message: unknown option --env')
    then('error includes hint: to pass args to your wish file, use: -- --env')
      sothat('user learns the correct syntax')

given('user typos a declastruct flag')
  when('user runs: npx declastruct plan --wishe resources.ts --into plan.json')
    then('declastruct errors with message: unknown option --wishe')
    then('error includes hint: to pass args to your wish file, use: -- --wishe')
      sothat('typos are caught instead of silent passthrough')

---

# usecase.6 = edge cases

given('wish file does not parse args')
  when('user runs: npx declastruct plan --wish simple.ts --into plan.json')
    then('declastruct succeeds')
    then('wish file sees empty args in process.argv')
      sothat('backwards compatibility is preserved')

given('user passes a flag that matches declastruct flag name')
  when('user runs: npx declastruct plan --wish resources.ts --into plan.json -- --wish custom-value')
    then('--wish custom-value appears in process.argv for wish file')
      sothat('user owns their arg namespace after --')

given('user passes boolean flags')
  when('user runs: npx declastruct plan --wish resources.ts --into plan.json -- --verbose --debug')
    then('wish file sees --verbose and --debug in process.argv')
      sothat('boolean flags pass through correctly')
